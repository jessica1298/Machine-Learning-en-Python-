np.random.seed(20200327)
x1 = np.random.normal(1, 0.1, 100)
x2 = np.random.normal(1, 0.1, 100) 
x1[0:30] = x1[0:30] + 1
x1[31:60] = x1[31:60] - 1
x2[0:30] = x2[0:30] + 1
x2[31:60] = x2[31:60] - 1
y = [1 if i < 60 else 0 for i in range(0,100)]
df = pd.DataFrame({"x1": x1,"x2": x2,"y" : y})
